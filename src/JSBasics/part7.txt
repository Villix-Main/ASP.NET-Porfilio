const developmentType = "web";

const documentation = document.querySelector("#documentation")

if (developmentType == "web") {
    documentation.setAttribute("href", "https://developer.mozilla.org/en-US/");
}
else if (developmentType == "csharp") {
    documentation.setAttribute("href", "https://docs.microsoft.com/en-us/");
}



const mainHeader = document.querySelector("h1");
mainHeader.innerText += ", We are learning about the DOM";

// const name = document.getElementById("name");
// name.innerText = "Josh";

const responses = document.querySelectorAll(".response");
responses.forEach(resp => {
    resp.innerText = "This is a new type of error";
    resp.style.color = "red";
    resp.style.fontSize = "20px";
    resp.style.fontWeight = "bold";
    resp.style.marginLeft = "20px";
    resp.style.fontStyle = "italic";
    resp.style.fontStyle = "";
});

const content = document.querySelector(".content");
content.style.backgroundColor = "#3f9faa";
content.innerHTML += "<p>Hello this content is generated by the DOM</p>"

const names = ["Dominic", "Jack", "Joe", "Max"];
names.forEach(name => {
     content.innerHTML += `<p>${name}</p>`
});

const codeDescription = document.querySelector(".codeDescription");

if (developmentType == "web") {
    codeDescription.classList.add("web");    
}
else if (developmentType == "csharp") {
    codeDescription.classList.add("csharp");
}

// This is the reason why the web class style is not being applied
codeDescription.classList.remove("web");


const paras = document.querySelectorAll("p");
paras.forEach(p => {
   if (p.textContent.includes("web")) {
       p.classList.add("web-text");
   } 
   else if (p.textContent.includes("csharp")) {
       p.classList.add("csharp-text");
   }
});
